require 'net/http'

class Scraper
  attr_reader :webpage

  def initialize(webpage = 'http://sethvargo.com')
    @webpage = URI.parse(webpage)
    write Net::HTTP.get(@webpage)
  end

  def write(contents)
    File.open("#{@webpage.host}.html", 'w') do |file|
      file.write(contents)
    end
  end
end

Scraper.new
